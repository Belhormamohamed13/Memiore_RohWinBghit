\section*{Annexe E — Manuel utilisateur}
\addcontentsline{toc}{section}{Annexe E — Manuel utilisateur}
\label{annexe:manuel}

\subsection*{Pour le Passager}
\begin{enumerate}
  \item Téléchargez l'application mobile RohWinBghit.
  \item Inscrivez-vous avec votre numéro de téléphone et validez le code OTP reçu par SMS.
  \item Effectuez la vérification de votre identité (capture de votre carte d'identité nationale et selfie de vivacité).
  \item Recherchez un trajet en spécifiant la wilaya de départ, la wilaya d'arrivée et la date.
  \item Réservez et payez en ligne ou choisissez le paiement en espèces.
\end{enumerate}

\subsection*{Pour le Conducteur}
\begin{enumerate}
  \item Basculez en mode conducteur depuis votre profil.
  \item Ajoutez les informations de votre véhicule (marque, modèle, année, photo d'immatriculation).
  \item Publiez un trajet en définissant les villes de passage, le nombre de places et le prix suggéré.
  \item Validez les demandes de réservation des passagers.
\end{enumerate}
